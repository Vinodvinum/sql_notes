Phase 1: Query Basics

Foundation - must master

SELECT

Syntax

SELECT column1, column2
FROM table_name;

Example

SELECT customer_id, company_name
FROM customers;

Use only required columns instead of SELECT * in interviews.

WHERE

Syntax

SELECT column_list
FROM table_name
WHERE condition;

Example

SELECT customer_id, company_name, city
FROM customers
WHERE city = 'London';

String values must be in single quotes.

INSERT

Syntax

INSERT INTO table_name (col1, col2, col3)
VALUES (val1, val2, val3);

Example

INSERT INTO departments (department_name, location)
VALUES ('Cardiology', 'Block A');

Column count must match value count.

UPDATE

Syntax

UPDATE table_name
SET col1 = value1, col2 = value2
WHERE condition;

Example

UPDATE patients
SET city = 'Mysore'
WHERE patient_id = 101;

Always add WHERE unless you intend to update all rows.

DELETE

Syntax

DELETE FROM table_name
WHERE condition;

Example

DELETE FROM temp_orders
WHERE order_date < '2021-01-01';

Without WHERE, every row in the table is deleted.

What You Should Achieve

  • Read and filter exact rows
  • Insert, update, and delete safely
  • Avoid dangerous no-WHERE mistakes

Phase 2: Query Filtering

Real analysis starts here

Operators (=, !=, >, <)

Syntax

WHERE col = value
WHERE col != value
WHERE col > value
WHERE col < value

Example

SELECT product_name, unit_price
FROM products
WHERE unit_price > 50;

ORDER BY

Syntax

SELECT col1, col2
FROM table_name
ORDER BY col1 ASC, col2 DESC;

Example

SELECT order_id, customer_id, order_date
FROM orders
ORDER BY order_date DESC;

LIKE

Syntax

WHERE name LIKE 'A%'
WHERE name LIKE '%son'
WHERE name LIKE '%ann%'

Example

SELECT first_name, last_name
FROM employees
WHERE last_name LIKE 'S%';

IN and BETWEEN

Syntax

WHERE city IN ('London', 'Paris', 'Berlin')
WHERE amount BETWEEN 1000 AND 5000

Example

SELECT order_id, customer_id, freight
FROM orders
WHERE freight BETWEEN 30 AND 80;

SELECT customer_id, city
FROM customers
WHERE city IN ('London', 'Madrid');

JOIN

Syntax

SELECT a.col1, b.col2
FROM table_a a
JOIN table_b b ON a.key = b.key;

Example

SELECT o.order_id, c.company_name, o.order_date
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id;

UNION and UNION ALL

Syntax

SELECT col1, col2 FROM table_a
UNION
SELECT col1, col2 FROM table_b;

SELECT col1, col2 FROM table_a
UNION ALL
SELECT col1, col2 FROM table_b;

Example

SELECT city FROM customers
UNION
SELECT city FROM suppliers;

UNION removes duplicates, UNION ALL keeps duplicates.

GROUP BY

Syntax

SELECT group_col, COUNT(*)
FROM table_name
GROUP BY group_col;

Example

SELECT city, COUNT(*) AS total_customers
FROM customers
GROUP BY city;

HAVING

Syntax

SELECT group_col, SUM(metric)
FROM table_name
GROUP BY group_col
HAVING SUM(metric) > value;

Example

SELECT customer_id, COUNT(*) AS total_orders
FROM orders
GROUP BY customer_id
HAVING COUNT(*) > 10;

CASE

Syntax

CASE
  WHEN condition1 THEN result1
  WHEN condition2 THEN result2
  ELSE result3
END AS alias_name

Example

SELECT product_name, unit_price,
CASE
  WHEN unit_price >= 80 THEN 'Premium'
  WHEN unit_price >= 40 THEN 'Standard'
  ELSE 'Budget'
END AS price_band
FROM products;

DISTINCT

Syntax

SELECT DISTINCT col1, col2
FROM table_name;

Example

SELECT DISTINCT country
FROM customers
ORDER BY country;

EXISTS

Syntax

SELECT *
FROM parent p
WHERE EXISTS (
  SELECT 1
  FROM child c
  WHERE c.parent_id = p.id
);

Example

SELECT c.customer_id, c.company_name
FROM customers c
WHERE EXISTS (
  SELECT 1
  FROM orders o
  WHERE o.customer_id = c.customer_id
);

ANY and ALL

Syntax

WHERE value > ANY (subquery)
WHERE value > ALL (subquery)

Example

SELECT product_name, unit_price
FROM products
WHERE unit_price > ALL (
  SELECT unit_price
  FROM products
  WHERE category_id = 2
);

NULL Handling

Syntax

WHERE col IS NULL
WHERE col IS NOT NULL
COALESCE(col, replacement)

Example

SELECT patient_id, COALESCE(allergies, 'NKA') AS allergy_status
FROM patients
WHERE allergies IS NULL OR allergies IS NOT NULL;

Aliases

Syntax

SELECT col AS alias_name
FROM table_name t;

Example

SELECT o.order_id AS order_no,
       c.company_name AS customer_name
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id;

What You Should Achieve

  • Join and filter datasets like real analyst tasks
  • Summarize metrics correctly with GROUP BY and HAVING
  • Write reusable query patterns for interviews

Phase 3: Functions

Advanced analyst mode

Aggregate Functions

Syntax

COUNT(*), SUM(col), AVG(col), MAX(col), MIN(col)

Example

SELECT customer_id,
       COUNT(*) AS total_orders,
       SUM(freight) AS total_freight,
       AVG(freight) AS avg_freight
FROM orders
GROUP BY customer_id;

Window Functions (LAG, LEAD, FIRST_VALUE, LAST_VALUE)

Syntax

FUNC(col) OVER (
  PARTITION BY group_col
  ORDER BY sort_col
)

Example

SELECT employee_id,
       order_date,
       freight,
       LAG(freight) OVER (PARTITION BY employee_id ORDER BY order_date) AS prev_freight,
       LEAD(freight) OVER (PARTITION BY employee_id ORDER BY order_date) AS next_freight,
       FIRST_VALUE(freight) OVER (PARTITION BY employee_id ORDER BY order_date) AS first_freight
FROM orders;

String Functions

Syntax

CONCAT(str1, str2)
LENGTH(str)
UPPER(str)
LOWER(str)

Example

SELECT employee_id,
       UPPER(last_name) AS last_name_upper,
       LOWER(first_name) AS first_name_lower,
       LENGTH(first_name) AS name_len
FROM employees;

Numeric Functions

Syntax

ROUND(num, digits)
FLOOR(num)
CEIL(num)
ABS(num)
POWER(a, b)
SQRT(num)

Example

SELECT product_id,
       unit_price,
       ROUND(unit_price, 1) AS rounded_price,
       FLOOR(unit_price) AS floor_price,
       CEIL(unit_price) AS ceil_price,
       POWER(unit_price, 2) AS square_price
FROM products;

Date Functions

Syntax

CURRENT_TIMESTAMP
EXTRACT(YEAR FROM date_col)
EXTRACT(MONTH FROM date_col)
EXTRACT(DAY FROM date_col)

Example

SELECT order_id,
       order_date,
       EXTRACT(YEAR FROM order_date) AS order_year,
       EXTRACT(MONTH FROM order_date) AS order_month,
       CURRENT_TIMESTAMP AS run_time
FROM orders;

What You Should Achieve

  • Perform advanced business calculations
  • Use windows for trend and comparison analysis
  • Handle text, numbers, and dates in one query

Phase 4: Tables

Database design and control

Data Types

Common Types

INT, BIGINT
VARCHAR(n), TEXT
DECIMAL(p,s)
DATE, TIMESTAMP
BOOLEAN

Choose type based on data meaning, range, and precision.

CREATE TABLE

Syntax

CREATE TABLE table_name (
  id INT PRIMARY KEY,
  col1 VARCHAR(100),
  col2 DECIMAL(10,2)
);

Example

CREATE TABLE appointments (
  appointment_id INT PRIMARY KEY,
  patient_id INT NOT NULL,
  doctor_id INT NOT NULL,
  appointment_date DATE NOT NULL,
  fee DECIMAL(10,2) DEFAULT 0
);

ALTER TABLE

Syntax

ALTER TABLE table_name ADD COLUMN col_name datatype;
ALTER TABLE table_name DROP COLUMN col_name;
ALTER TABLE table_name ALTER COLUMN col_name TYPE datatype;

Example

ALTER TABLE appointments
ADD COLUMN notes VARCHAR(255);

DROP TABLE

Syntax

DROP TABLE table_name;

Example

DROP TABLE old_appointments_backup;

Constraints

Syntax

NOT NULL
UNIQUE
PRIMARY KEY
FOREIGN KEY
CHECK
DEFAULT

Example

CREATE TABLE departments (
  department_id INT PRIMARY KEY,
  department_name VARCHAR(80) UNIQUE NOT NULL,
  location VARCHAR(80) DEFAULT 'Main Block'
);

CREATE TABLE doctors (
  doctor_id INT PRIMARY KEY,
  department_id INT,
  salary DECIMAL(10,2) CHECK (salary > 0),
  FOREIGN KEY (department_id) REFERENCES departments(department_id)
);

AUTO INCREMENT and INDEX

Syntax

id INT GENERATED ALWAYS AS IDENTITY
CREATE INDEX idx_name ON table_name (col1, col2);

Example

CREATE TABLE visits (
  visit_id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  patient_id INT NOT NULL,
  visit_date DATE NOT NULL
);

CREATE INDEX idx_orders_customer_date
ON orders (customer_id, order_date);

What You Should Achieve

  • Create robust schema with constraints
  • Change schema safely using ALTER
  • Use indexes for better query performance

Final Strategy (How To Learn Without Getting Lost)

Step 1 Phase 1 for 2-3 days. Build clean basics and safe update/delete habits.
Step 2 Phase 2 for 5-7 days. Join, Group By, and Having are highest priority.
Step 3 Phase 3 for 5 days. Focus on windows and date logic for real analytics.
Step 4 Phase 4 for 2-3 days. Understand schema, keys, and indexing basics.
Reality Check For analyst interviews: JOIN, GROUP BY, HAVING, and window functions are asked most.
Proceed Order Query Basics -> Query Filtering -> Functions -> Tables.

Explain Like Real (Speaking Script)

Single Table Question

I am using the table shown in schema, selecting only requested columns, applying WHERE conditions to filter rows, and then sorting if needed for final presentation.

JOIN Question

I start from the fact table, join related lookup tables using key columns from schema, then apply filters and return business-friendly output columns.

Aggregation Question

I group by the requested business dimension, compute aggregate metrics like COUNT or SUM, and use HAVING only when filtering grouped results.

Top N Question

I calculate the metric first, sort descending to rank highest values, and use LIMIT to return the top requested records.
