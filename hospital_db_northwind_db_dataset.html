Northwind DB

Interview-style Northwind query set.

Easy (1-9)

1) Show category_name and description sorted by category_name.

SELECT category_name, description
FROM categories
ORDER BY category_name;

2) Customers not from Germany, Mexico, Spain.

SELECT contact_name, address, city
FROM customers
WHERE country NOT IN ('Germany', 'Mexico', 'Spain');

3) Orders placed on 2018-02-26.

SELECT order_date, shipped_date, customer_id, freight
FROM orders
WHERE order_date = '2018-02-26';

4) Orders shipped later than required date.

SELECT employee_id, order_id, customer_id, required_date, shipped_date
FROM orders
WHERE shipped_date > required_date;

5) Even numbered order_id values.

SELECT order_id
FROM orders
WHERE order_id % 2 = 0;

6) Customers with letter L in city, sorted by contact_name.

SELECT city, company_name, contact_name
FROM customers
WHERE city LIKE '%L%'
ORDER BY contact_name;

7) Customers that have fax numbers.

SELECT company_name, contact_name, fax
FROM customers
WHERE fax IS NOT NULL;

8) Most recently hired employee.

SELECT first_name, last_name, hire_date
FROM employees
ORDER BY hire_date DESC
LIMIT 1;

9) Average unit price, total stock, and total discontinued products.

SELECT
  ROUND(AVG(unit_price), 2) AS avg_price,
  SUM(units_in_stock) AS total_stock,
  SUM(discontinued) AS total_discontinued
FROM products;
Medium (10-13)

10) Product, supplier, and category names.

SELECT
  p.product_name,
  s.company_name,
  c.category_name
FROM products p
JOIN suppliers s ON p.supplier_id = s.supplier_id
JOIN categories c ON p.category_id = c.category_id;

11) Average unit price by category.

SELECT
  c.category_name,
  ROUND(AVG(p.unit_price), 2) AS avg_price
FROM products p
JOIN categories c ON p.category_id = c.category_id
GROUP BY c.category_name;

12) Merge customers and suppliers with source column.

SELECT city, company_name, contact_name, 'customers' AS source
FROM customers
UNION
SELECT city, company_name, contact_name, 'suppliers' AS source
FROM suppliers;

13) Total orders by year and month.

SELECT
  strftime('%Y', order_date) AS year,
  strftime('%m', order_date) AS month,
  COUNT(*) AS total_orders
FROM orders
GROUP BY year, month;
Hard (14-15)

14) Employee order count with shipping status (On Time, Late, Not Shipped).

SELECT
  e.first_name,
  e.last_name,
  COUNT(o.order_id) AS num_orders,
  CASE
    WHEN o.shipped_date IS NULL THEN 'Not Shipped'
    WHEN o.shipped_date <= o.required_date THEN 'On Time'
    ELSE 'Late'
  END AS shipped_status
FROM employees e
JOIN orders o ON e.employee_id = o.employee_id
GROUP BY e.employee_id, e.first_name, e.last_name, shipped_status
ORDER BY e.last_name, e.first_name, num_orders DESC;

15) Discount loss per year (most recent first).

SELECT
  strftime('%Y', o.order_date) AS year,
  ROUND(SUM(od.unit_price * od.quantity * od.discount), 2) AS total_loss
FROM orders o
JOIN order_details od ON o.order_id = od.order_id
GROUP BY year
ORDER BY year DESC;

Hospital DB

Deduplicated hospital query set from the first complete block in practice2.txt.

Easy (1-16)

1) Male patients.

SELECT first_name, last_name, gender
FROM patients
WHERE gender = 'M';

2) Patients with no allergies (NULL).

SELECT first_name, last_name
FROM patients
WHERE allergies IS NULL;

3) First names starting with C.

SELECT first_name
FROM patients
WHERE first_name LIKE 'C%';

4) Patients with weight 100 to 120.

SELECT first_name, last_name
FROM patients
WHERE weight BETWEEN 100 AND 120;

5) Update NULL allergies to NKA.

UPDATE patients
SET allergies = 'NKA'
WHERE allergies IS NULL;

6) Full name concatenation.

SELECT first_name || ' ' || last_name AS full_name
FROM patients;

7) Patient name with full province name.

SELECT p.first_name, p.last_name, pn.province_name
FROM patients p
JOIN province_names pn ON p.province_id = pn.province_id;

8) Count patients born in 2010.

SELECT COUNT(*) AS total
FROM patients
WHERE strftime('%Y', birth_date) = '2010';

9) Tallest patient.

SELECT first_name, last_name, height
FROM patients
ORDER BY height DESC
LIMIT 1;

10) Patients by patient_id list.

SELECT *
FROM patients
WHERE patient_id IN (1, 45, 534, 879, 1000);

11) Total number of admissions.

SELECT COUNT(*) AS total_admissions
FROM admissions;

12) Admissions with same-day admit and discharge.

SELECT *
FROM admissions
WHERE admission_date = discharge_date;

13) Admissions count for patient 579.

SELECT patient_id, COUNT(*) AS total
FROM admissions
WHERE patient_id = 579
GROUP BY patient_id;

14) Unique cities in province NS.

SELECT DISTINCT city
FROM patients
WHERE province_id = 'NS';

15) Patients with height > 160 and weight > 70.

SELECT first_name, last_name, birth_date
FROM patients
WHERE height > 160 AND weight > 70;

16) Patients from Hamilton with allergies.

SELECT first_name, last_name, allergies
FROM patients
WHERE allergies IS NOT NULL
  AND city = 'Hamilton';
Medium (17-30)

17) Unique birth years sorted ascending.

SELECT DISTINCT strftime('%Y', birth_date) AS year
FROM patients
ORDER BY year;

18) First names that appear only once.

SELECT first_name
FROM patients
GROUP BY first_name
HAVING COUNT(*) = 1;

19) Names starting and ending with s, length at least 6.

SELECT patient_id, first_name
FROM patients
WHERE first_name LIKE 's%s'
  AND LENGTH(first_name) >= 6;

20) Patients diagnosed with Dementia.

SELECT p.patient_id, p.first_name, p.last_name
FROM patients p
JOIN admissions a ON p.patient_id = a.patient_id
WHERE a.diagnosis = 'Dementia';

21) First names ordered by length, then alphabetically.

SELECT first_name
FROM patients
ORDER BY LENGTH(first_name), first_name;

22) Male and female patient counts in one row.

SELECT
  SUM(CASE WHEN gender = 'M' THEN 1 ELSE 0 END) AS male,
  SUM(CASE WHEN gender = 'F' THEN 1 ELSE 0 END) AS female
FROM patients;

23) Patients allergic to Penicillin or Morphine.

SELECT first_name, last_name, allergies
FROM patients
WHERE allergies IN ('Penicillin', 'Morphine')
ORDER BY allergies, first_name, last_name;

24) Repeated diagnosis admissions by patient.

SELECT patient_id, diagnosis
FROM admissions
GROUP BY patient_id, diagnosis
HAVING COUNT(*) > 1;

25) Patient count per city.

SELECT city, COUNT(*) AS total
FROM patients
GROUP BY city
ORDER BY total DESC, city;

26) Patients and doctors combined with role label.

SELECT first_name, last_name, 'Patient' AS role
FROM patients
UNION
SELECT first_name, last_name, 'Doctor' AS role
FROM doctors;

27) Popular allergies excluding NULL and NKA.

SELECT allergies, COUNT(*) AS total
FROM patients
WHERE allergies IS NOT NULL
  AND allergies != 'NKA'
GROUP BY allergies
ORDER BY total DESC;

28) Patients born in the 1970s.

SELECT first_name, last_name, birth_date
FROM patients
WHERE birth_date BETWEEN '1970-01-01' AND '1979-12-31'
ORDER BY birth_date;

29) Name format LASTNAME,firstname.

SELECT UPPER(last_name) || ',' || LOWER(first_name) AS full_name
FROM patients
ORDER BY first_name DESC;

30) Provinces with total patient height at least 7000.

SELECT province_id, SUM(height) AS total_height
FROM patients
GROUP BY province_id
HAVING total_height >= 7000;
Hard (31-35)

31) Weight difference for last name Maroni.

SELECT MAX(weight) - MIN(weight) AS difference
FROM patients
WHERE last_name = 'Maroni';

32) Admission count by day of month.

SELECT strftime('%d', admission_date) AS day, COUNT(*) AS total
FROM admissions
GROUP BY day
ORDER BY total DESC;

33) Latest admission for patient 542.

SELECT *
FROM admissions
WHERE patient_id = 542
ORDER BY admission_date DESC
LIMIT 1;

34) Patients with no admissions.

SELECT *
FROM patients p
WHERE NOT EXISTS (
  SELECT 1
  FROM admissions a
  WHERE a.patient_id = p.patient_id
);

35) Patient name, doctor name, and diagnosis.

SELECT
  p.first_name || ' ' || p.last_name AS patient_name,
  d.first_name || ' ' || d.last_name AS doctor_name,
  a.diagnosis
FROM admissions a
JOIN patients p ON a.patient_id = p.patient_id
JOIN doctors d ON a.attending_doctor_id = d.doctor_id;
